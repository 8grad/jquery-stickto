This is an example of a composition of two stickies, one inside the other. They are both dynamic, updating their contents as you scroll. The second (smaller) text box is horizontally aligned with its container, even though it is "stuck" to this text field. 

$('.stick.four.cont').stickTo({
	target: '.sticker.four',
    margin: "0 10",
    align: 'outside-left',
    scrollIn : function() { this.html('SCROLLED IN!'); },
    scrollOut: function() { this.html('SCROLLED OUT!'); },
    onScroll: function(pct, data) {
        // This code does some coloring based on scroll percentage
        this.css({
            'background-color': 'rgb(0, '+parseInt(pct*255)+', 0)',
            'color': 'rgb(100, '+parseInt(255*(1-pct))+', 50)'
        });
    }
}).css('height', $(window).height());


$('.stick.four.slider').stickTo({
    target: '.sticker.four',
    margin: "0",
    align: 'center',
    alignWith: '.stick.four.cont',
    onScroll : function(pct, data) {
        this.html("Scrolled "+(100*pct)+"%");
    }
}).css('height', '40px');
